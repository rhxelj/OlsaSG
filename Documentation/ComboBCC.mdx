# ComboBCC

Este componente dibuja tres botones Borrar, Confirmar, Cancelar y recibe mediante props el onClick event como props.actions.borrar, props.actions.confirmar, props.actions.cancelar

```
import React from "react";
import { Button, Paper } from "@material-ui/core";

export default function ComboBCC(props: any) {
  return (
    <>
      <Button
        onClick={props.actions.borrar}
        style={{ margin: ".5rem" }}
        variant="contained"
      >
        Borrar
      </Button>
      <Button
        onClick={props.actions.confirmar}
        style={{ margin: ".5rem" }}
        variant="contained"
        color="primary"
      >
        Confirmar
      </Button>
      <Button
        onClick={props.actions.cancelar}
        style={{ margin: ".5rem" }}
        variant="contained"
        color="secondary"
      >
        Cancelar
      </Button>
    </>
  );
}
```
